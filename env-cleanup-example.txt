# File Cleanup Configuration
# Add these variables to your .env file

# Enable/Disable file cleanup system
FILE_CLEANUP_ENABLED=true

# Automatic cleanup when models are deleted
FILE_AUTO_CLEANUP_ON_DELETE=true

# Scheduled cleanup configuration
SCHEDULED_CLEANUP_ENABLED=true
SCHEDULED_CLEANUP_FREQUENCY=daily  # daily, weekly, monthly
SCHEDULED_CLEANUP_TIME=02:00       # 24-hour format
SCHEDULED_CLEANUP_DISK=public
SCHEDULED_CLEANUP_BATCH_SIZE=100

# File retention period (days)
FILE_RETENTION_DAYS=7

# Logging configuration
FILE_CLEANUP_LOGGING_ENABLED=true
FILE_CLEANUP_LOG_CHANNEL=file_cleanup
FILE_CLEANUP_LOG_LEVEL=info
FILE_CLEANUP_LOG_RETENTION_DAYS=30

# Safety settings
CLEANUP_MAX_FILES_PER_RUN=1000
CLEANUP_REQUIRE_CONFIRMATION=true
CLEANUP_BACKUP_BEFORE_DELETE=false
CLEANUP_BACKUP_DISK=local

# Admin email for failure notifications
MAIL_ADMIN_EMAIL=admin@yourdomain.com
